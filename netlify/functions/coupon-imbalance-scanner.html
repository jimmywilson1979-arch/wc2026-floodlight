// netlify/functions/coupon.js // Oddschecker blocks direct Lambda fetches at Cloudflare (403). // Route through Jina Reader (free, renders from non-flagged IPs) to get past it. // Returns JSON: { ok, status, length, matches_seen, html } for the front-end. exports.handler = async () => { const coupon = 'https://www.oddschecker.com/football/bumper-coupon'; // Jina Reader proxy — fetches + renders the page, returns markdown/text. const target = 'https://r.jina.ai/' + coupon + '?_=' + Date.now(); const cors = { 'access-control-allow-origin': '*', 'cache-control': 'no-store', 'content-type': 'application/json; charset=utf-8', }; try { const r = await fetch(target, { headers: { 'User-Agent': 'Mozilla/5.0 (Linux; Android 14; Pixel 8) AppleWebKit/537.36 ' + '(KHTML, like Gecko) Chrome/126.0.0.0 Mobile Safari/537.36', Accept: 'text/plain, text/markdown, */*', 'Accept-Language': 'en-GB,en;q=0.9', 'X-No-Cache': 'true', }, }); const html = await r.text(); const hits = (html.match(/View all .+? v .+? odds/g) || []).length; return { statusCode: 200, headers: cors, body: JSON.stringify({ ok: r.ok, status: r.status, length: html.length, matches_seen: hits, html, }), }; } catch (e) { return { statusCode: 200, headers: cors, body: JSON.stringify({ ok: false, error: String((e && e.message) || e) }), }; } };
